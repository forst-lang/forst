```forst
package main

type PlaceOrderInput = {
	stockKeepingUnit: String.Min(1).Max(64),
	quantity:         Int.Min(1).Max(99),
}

func placeOrder(in: PlaceOrderInput) {
	println("Order placed:", in.stockKeepingUnit, in.quantity)
}

func main() {
	placeOrder({
		stockKeepingUnit: "ITEM-1",
		quantity:         2,
	})
}
```
