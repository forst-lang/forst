    type Logger = {
    	info(msg String)
    }

    type Clock = {
    	now(): Int
    }

    func logExpiry(id String) {
    	use logger: Logger
    	logger.info("expire " + id)
    }

    func expireToken(token Token) {
    	use clock: Clock
    	if token.expiresAt < clock.now() {
    		logExpiry(token.id)
    	}
    }
